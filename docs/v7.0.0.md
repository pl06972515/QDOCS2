>     console.log(items[i]);
>}
>
>for (let x of items) {
>     console.log(x);
>}
>
>for (let [key, value] of Object.entries(items)) {
>
>     // 0 ：1
>     // 1 ：2
>     // 2 ：3
>     console.log(`${key}: ${value}`);
>}
>
>----------------------------------------
>
># [ ES6 ] 展开运算符
>let items = [1, 2, 3, 4, 5]

>console.log(Math.max(...items)) // 5
