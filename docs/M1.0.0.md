#### **[ M1.js ]**

!> <span style='color:red'>[ 隔离性：在`ES6`，浏览器不会将模块内代码包装函数，直接使用模块作用域 ]</span>

```csharp
# [ 命名导出 ]
export const x = 100;
export const y = 500;

function M1() { 
    x = 1000;
    return x;
}

// export {
//     x,
//     y,
//     M1
// }


```



#### **[ ℹ️ Main.js ]**

!> <span style='color:red'>[`ES6`模块对象属性是只读的 ] [`ES6`命名导入的变量 → 与原模块变量共用同一块内存 ]</span>

```csharp
# [ 命名导入 ]
import { x as v1, y, M1 } from "./M1.js"
M1();

import { x as v1 } from "./M1.js"
console.log(v1);  // 1000


```

```csharp
# [ 动态导入 ] Promise ( 浏览器加载完 JS 后 - 回调 )
const p = import("./M1.js")
p.then(module => {

    console.log(module.x);
});


```
