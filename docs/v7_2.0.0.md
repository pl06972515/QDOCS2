>```csharp
>const items = new Map();
>const items = new Map([['key1', 'value1'], ['key1', 'value2']]); 
>
># [ 循环迭代 ]
>for (const [key, value] of items) {
>  
>}
>
>    
>```
>
>
>
><br/>

```csharp
{
    
     # [ 常用函数 ]
      
     items.size
         
     items.set('name', 'John');
     items.get('name')
         
     v = items.delete('name') // false: 删除失败( 因键不存在 )\
     items.clear()
         
     items.has('name')        // [ 检测 ] 是否存在指定的 KEY
     items.keys()
     items.values()
         
}
```
