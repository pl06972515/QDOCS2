#### **[ 构造函数 ]**

```csharp
class TClass
{

    constructor(x, y)
    {
        this.x = x;
        this.y = y;
    }

    // ES6: [ 普通函数 ] 自动添加到原型上 TClass.prototype
    print()
    {
        console.log(`x: ${this.x}, y: ${this.y}`);
    }

    static init()
    {

        console.log('TClass init');
    }

}


```

```csharp
const obj = new TClass(1, 2);
console.log(obj.__proto__ === TClass.prototype);   // true
console.log(obj.print === TClass.prototype.print); // true
obj.print();

TClass.init();


```



#### **[ ℹ️ 类型继承 ]**

```csharp
class BaseClass
{

    constructor(x, y)
    {
        this.x = x;
        this.y = y;
    }

    print()
    {
        console.log(`x: ${this.x}, y: ${this.y}`);
    }

}


```

```csharp
class TClass extends BaseClass
{

    constructor(x, y, z)
    {
        super(x, y);
        this.z = z;
    }

    print()
    {
        super.print();
        console.log(`x: ${this.x}, y: ${this.y}, z: ${this.z}`);
    }

}


```

```csharp
const obj = new TClass(1, 2, 3);
console.log(obj.__proto__ === TClass.prototype);                 // true
console.log(TClass.prototype.__proto__ === BaseClass.prototype); // true
console.log(BaseClass.prototype.__proto__ === Object.prototype); // true


```
