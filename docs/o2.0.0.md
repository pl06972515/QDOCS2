#### **[ 构造函数 ]**

```csharp
class TClass
{

     constructor(x, y)
     {
          this.x = x;
          this.y = y;
     }

     # [ 普通函数 ] 自动定义到 TClass 原型对象上 ( TClass.prototype )
     M1()
     {
          console.log(`x: ${this.x}, y: ${this.y}`);
     }

     # [ 静态函数 ]
     static Init()
     {

          console.log('TClass init');
     }

}


```

```csharp
# o.__proto__ === TClass.prototype : true
# o.M1 === TClass.prototype.M1: true
    
const o = new TClass(1, 2);
o.M1();

TClass.Init();


```



#### **[ ℹ️ 类型继承 ]**

```csharp
class BaseClass
{

     constructor(x, y)
     {
          this.x = x;
          this.y = y;
     }

     print()
     {
          console.log(`x: ${this.x}, y: ${this.y}`);
     }

}


```

```csharp
class TClass extends BaseClass
{

     # [ 调用基类构造 ] super() 等效 base
     constructor(x, y, z)
     {
          super(x, y);
          this.z = z;
     }

     print()
     {
          super.print();
          console.log(`x: ${this.x}, y: ${this.y}, z: ${this.z}`);
     }

}


```

```csharp
# v.__proto__ === TClass.prototype : true
# TClass.prototype.__proto__ === BaseClass.prototype : true
# BaseClass.prototype.__proto__ === Object.prototype : true
    
const v = new TClass(1, 2, 3);


```
