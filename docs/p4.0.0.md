#### **[ 捕获异常 ]**

```csharp
function fn()
{

     try{

         throw new Error("Test error with value: " + x);

     } catch(err) {
    
         console.log(err.message);
         throw new Error("Error from finally block");  // return;
    
     } finally {

         console.error("In finally block");
     }

}

fn();


```



#### **[ 抛出异常 ]**

```csharp
function fn(x)
{
    
    if (x > 5)
        throw new Error(`x 不能大于 5。x 的值为: ${ 10 }`);
    
}

fn(10);


```
