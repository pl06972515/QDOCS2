#### **操作运算**

| 运算符类型                                |                                                              |
| ----------------------------------------- | ------------------------------------------------------------ |
| 算术运算符                                | `+  -  *  /  %`                                              |
| 一元运算符                                | `++ --`<span style='color:red'> [`++`在后：先运算在`++`] [`++`在前：先`++`在运算 ]</span> |
| 关系运算符                                | <span style='color:red'>[`===`严格相等：比较值和类型，不进行类型转换 ]</span>`!=  >  >=  <  <=` |
| 逻辑运算符                                | `&& \|\| !( 取反操作类似: !true  )`<span style='color:red'>[ 特殊：当左侧为`false`值时返回右侧`const x = (false \|\| 'Default')`]</span> |
| <span style='color:red'>三元运算符</span> | `..?..:..`                                                   |

```csharp
# [ 宽松匹配 ] ==
console.log(5 == "5")          // true ( 因类型转换："5" 被转换为数字 5 )
console.log(true == 1)         // true ( 因类型转换：bool 被转换为数字 1 )
console.log(null == undefined) // true
console.log("" == 0)           // true ( 空字符串被转换为 0 )

# [ 严格匹配 ] ===
console.log(5 === "5")          // false
console.log(true === 1)         // false
console.log(null === undefined) // false
console.log("" === 0)           // false


```





#### **条件运算  if switch**

```javascript
if (condition1) {
    // [ A ] condition1 = True
} else if (condition2) {
    // [ B ] condition1 = False && condition2 = True
} else {
    // [ C ] condition1 = False && condition2 = False
}

    
```

```javascript
# switch 字面匹配
let score = 5;
switch (value) {
    case 5:
        console.log("匹配到 5");
        break;
    case "5":
        console.log("匹配到字符串 5");
        break;
    default:
        console.log("没有匹配");
}


```

```javascript
# [ switch: true ] 守卫条件 Guards
switch (true) {
    case score >= 90 && score <= 100:
        console.log("优秀");
        break;
    case score >= 80:
        console.log("良好");
        break;
    case score >= 60:
        console.log("及格");
        break;
    default:
        console.log("不及格");
}


```



#### **迭代运算 for**

!> <span style='color:red'>支持运算符 [`continue, break`]</span>

```csharp
let count = 0;
while (count < 3) {
    console.log('count: ' + count);
    count++;
}
            

```



```javascript
for (let i = 0; i < 5; i++) {
    console.log('第 ' + i + ' 次循环');
}


```

```javascript
# [ for...of ] 遍历可迭代对象 iterable
const fruits = ['苹果', '香蕉', '橙子'];
for (let fruit of fruits) {
     console.log(fruit);
}
    
        
```
