>    
>}
>

#### **[ ℹ️ 异步迭代 ]**

```csharp
# [ 异步迭代 ]
async function* asyncGenerator() {

    const task1 = new Promise((resolve) => {
        setTimeout(() => resolve(100), 1000);
    });
    yield await task1;

    const task2 = new Promise((resolve) => {
        setTimeout(() => resolve(200), 1000);
    });
    yield await task2;

    const task3 = new Promise((resolve) => {
        setTimeout(() => resolve(300), 1000);
    });
    yield await task3;

}


```

```csharp
async function main() {
    
    for await (const task of asyncGenerator()) {
        console.log(task);
    }
    
}

await main();


```





#### **[ 🟢 封装 ] XMLHttpRequest**

```csharp
<script type="module">

    function requestAsync() {

        const task = new Promise((resolve, reject) => {

            const xhr = new XMLHttpRequest();
            xhr.open('GET', 'http://192.168.8.113:5000');
            xhr.addEventListener('loadend', function () {

                if (this.status >= 200 && this.status < 300) {
                    resolve(this.responseText);
                    return;
                }
                reject(new Error("请求失败，状态码：" + this.status));

            });
            xhr.send();

        });
        return task;

    }

    # 方式1: 使用 async/await
    document.getElementById('btnRequest').addEventListener('click', async function () {

        try {

            const result = await requestAsync();
            console.log('请求成功:', result);

        } catch (error) {
            console.error('请求失败:', error);
        }

    });

    # 方式2: 使用 Promise.then()
    document.getElementById('btnRequest2').addEventListener('click', function() {
        
        const task = requestAsync();
        task.then(result => {
                 console.log('请求成功:', result);
             })
             .catch(error => {
                 console.error('请求失败:', error);
             });
        
    });


</script>


```
