#### **void**

!> <span style='color:red'>针对无返回值函数 [ 实际上其返回了`undefined`]</span>

```csharp
let M1 = function () {

    console.log('M1 function')
}

console.log(typeof M1) // function
console.log(M1())      // undefined


```



#### **[ 参数 ] params**

>[!WARNING|style: flat|label: 简要说明 ]
>
>- 函数参数：[ 支持可选参数 ]
>
>  参数传递：[ 位置传参 ] [ 命名参数 ]
>
>
><br/>

```csharp
let M1 = function (x, y = 10, ...args) {

     # [ object ]
     console.log(typeof args) 
     for (let [key, value] of Object.entries(args)) {

         // 0: 30
         // 1: 40
         // 2: 50
         console.log(`${key}: ${value}`);

     }
     console.log(args[0]);

}

M1(x = 10, 20, 30, 40, 50)


```



#### **[ 解包 ] return**

```csharp
function M1() {
    
    let n1 = 10;
    let n2 = 20;
    let n3 = 30;

    # [ 返回对象 ] 使用简化语法 ( ES6+ )
    return { n1, n2, n3 };
    
}

// 10, 20, 30 ( 不支持 _ )
const { n1, n2, n3 } = M1();
console.log(n1, n2, n3); 


```





#### **[ ℹ️ lambda ]**

!> （`JS`) 使用`lambda`关键字，来创建匿名函数 

```csharp
// [ Action ]
M1 = lambda: print("Hello, world!")
M1()

// [ Action<int, int> ]
M1 = lambda x, y=30: print(x + y)
M1(10)


```

```csharp
// [ Func<int> ]
M1 = lambda : 100
x = M1()

// [ Func<int, int, int> ]
M1 = lambda x, y=30: x + y
x = M1(10)
    

```



#### **[ ℹ️ 立即执行 ]**

!> <span style='color:red'>[ 函数定义后马上执行，不需要单独调用 ]</span>

#### **[ ✡ 回调函数 ]**

```csharp
function M1() {
    return 'M1';
}

# callback: 回调函数 ( C# 委托 )
function Main(callback) {
    return callback();
}

const v = Main(M1);
console.log(v); // Should print 'M1'


```
