#### **void**

!> <span style='color:red'>针对无返回值函数 [ 实际上其返回了`undefined`]</span>

```csharp
let M1 = function () {

     console.log('M1 function')
}

console.log(typeof M1) // function
console.log(M1())      // undefined


```



#### **[ 函数参数 ]**

!> <span style='color:red'>[ 在`JS`中，只支持位置传递 ]</span>

```csharp
function M1({ Name, Age }) {
    console.log(`姓名: ${Name}, 年龄: ${Age}`);
}

# [ 展开传递 ] 
M1({ Name: "张三", Age: 30 })


```

```csharp
          // 1: 40
          // 2: 50

     }
     console.log(args[0]);

}

M1(10, 20, 30, 40, 50)
    
    
```

#### **[ ℹ️ 箭头函数 ] lambda **

>[!WARNING|style: flat|label: 简要说明 ]
>
>- <span style='color:red'>[ 箭头函数特殊，函数内部不绑定`this`；箭头函数内`this`由外层作用域确定，逐层向上查找最近作用域中的`this`]</span> [<span style='color:#008B00'>[👓 Q1 ]</span>](F2.0.0  ':target=_blank')
>
>   <span style='color:Blue'>[ 建议：配合匿名函数使用 ]</span>
>
>
><br/>

```csharp
// [ Action ]
const M1 = () => print("Hello, world!")
M1()

// [ Action<int, int> ]
const M1 = (x, y=30, ...args) => print(x + y)
M1(10)

------------------------------------
    
// [ Func<int> ]
const M1 = () => 100
x = M1()

// [ Func<int, int, int> ]
const M1 = (x, y=30) => { return x + y }
x = M1(10)


```



#### **[ ℹ️ 回调函数 ]**

!> <span style='color:red'>回调函数 ( 类似`C#`委托 ) [ 入参`callback`：可以传递`lambda`]</span>

```csharp
# [ 回调函数 ] callback
function Main(callback) {
    return callback(10, 20);
}

--------------------------------------
    
# [ A ] 标准函数
function M1(x, y) {
    return x + y
}

const v = Main(M1);
console.log(v);   // 30

# [ B ] 匿名函数 lambda
const v = Main((x, y) => x + y);
console.log(v);  // 30


```
