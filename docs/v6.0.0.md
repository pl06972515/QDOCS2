>```csharp
>let x = {
>    
>     Name: "李四",
>     Age: 30,
>     Print: function() {
>         const _this = this;
>         console.log(_this.Name + " is " + _this.Age + " years old.");
>     }
>     
>};
>
>console.log(typeof x);  // object
>console.log(x.Name);    // 李四
>console.log(x["Age"]);  // 30   
>x.Print()
>    
>
># [ 检测 ] 属性是否存在
>console.log(x.hasOwnProperty("Name")); // true
>console.log("Age" in x);               // true
>
># [ 遍历对象 ]
>for (let key in x) {
>
>      console.log(key + ": " + x[key]);
>}
>
>for (let [key, value] of Object.entries(x)) 
>{
>
>      console.log(`${ key}: ${ value}`);
>}
>
>
>```

```csharp
{
    
     # [ 常用函数 ]
    
     // [ 添加属性 ]
     x.Gender = "男";
     x["Hobby"] = "编程";
     
     // [ 修改属性 ]
     x.Name = "王五";
     x.Age = 25;
    
     // [ 删除属性 ]
     delete x.Age;
     delete x["Gender"];
      
     # [ 获取 ] 所有属性名
     console.log(Object.keys(x));    // ["Name", "Age", ...]
     # [ 获取 ] 所有属性值
     console.log(Object.values(x));  // ["李四", 30, ...]
    
     # [ 获取 ] 键值对数组 
     /// <summary>
     /// for (let [key, value] of Object.entries(x)) 
     /// {
     ///     console.log(`${ key}: ${ value}`);
     /// }
     /// 
     /// </summary>
     console.log(Object.entries(x)); // [["Name", "李四"], ["Age", 30], ...]
    
     
}
```
