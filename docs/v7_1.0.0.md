>```csharp

>

>
># [ A ][ 并集 ] 
>const arr1 = [1, 2, 3, 4];
>const arr2 = [3, 4, 5, 6];
>const union = [...new Set([...arr1, ...arr2])];
>    
># [ A ][ 交集 ]
>const arr1 = [1, 2, 3, 4];
>const arr2 = [3, 4, 5, 6];
>const intersection = [...new Set(arr1)].filter(x => new Set(arr2).has(x));
>
># [ B ][ 差集 ]
>const arr1 = [1, 2, 3, 4];
>const arr2 = [3, 4, 5, 6];
>const difference = [...new Set(arr1)].filter(x => !new Set(arr2).has(x));
>
>
>```
