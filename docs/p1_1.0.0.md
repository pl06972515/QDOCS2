#### **[ 数组解构 ]**

>[!WARNING|style: flat|label: 简要说明 ]
>
><span style='color:Blue'>[ 数组解构 ]</span> 是指将 [ 数组中的元素 → 分别赋值给多个变量 ]
>
>- <span style='color:red'>[ 宽松匹配 ] 不匹配的变量赋予`undefined | []`</span>
>- <span style='color:red'>[ 使用：`_`弃元模式，`...`获取剩余元素( 在`JS`中，需最后一个参数 ) ]</span>
>
><br/>

```csharp
const [x, _, z, ...vs] = [1, 2, 3, 4, 5];

console.log(x);   // 1
console.log(z);   // 3
console.log(vs);  // [ 4, 5 ] 在JS 中剩余元素必须放在最后


```

```csharp
# [ 案例 ] 交换变量值
let a = 1, b = 2;
[a, b] = [b, a];
console.log(a, b); 


```



#### **[ ✡ 对象解构 ]**

>[!WARNING|style: flat|label: 简要说明 ]
>
>- <span style='color:red'>基本解构 [ 要求：变量名称 = 属性名称 ] [ 变量重命名 - 对象属性: 变量名称 ]</span>
>
>```csharp
>const { Name: x, Age, ...args } = {
>
>     Name: "李四",
>     Age: 30,
>     Print: function () {
>         const _this = this;
>         console.log(_this.Name + " is " + _this.Age + " years old.");
>     }
>
>};
>
>// 李四 30
>console.log(x, Age);  
>
>
>```
>
>---
>
>
>
>- <span style='color:Blue'>[ 嵌套结构：多级解构 ]</span>
>
>```csharp
>const { Name: uname, Age, address: { city, zip } } = { 
>     Name: "李四", 
>     Age: 30, 
>     address: { city: "北京", zip: "100000" } 
>};
>
>console.log(`用户名: ${uname}, 年龄: ${Age}, 城市: ${city}, 邮政编码: ${zip}`);
>
>
>```
>
>---
>
>
>
>- <span style='color:red'>[ 嵌套结构：数组解构 ]</span>
>
>```csharp
>const items = [ 
>     { Name: "李四", Age: 30 },
>     { Name: "王五", Age: 25 } 
>];
>
># [ A ] 循环解构
>for (const { Name, Age } of items) {
>     console.log(`姓名: ${Name}, 年龄: ${Age}`);
>}
>
># [ B ] 循环解构 forEach
>items.forEach(({ Name, Age }) => {
>     console.log(`姓名: ${Name}, 年龄: ${Age}`);
>});
>
>
>```
>
>
>
