<br/>

!> <span style='color:RED'>[`null`表示`无`或`空`]</span>



```csharp
# ?? 空合并运算符
let x = null;
let y = x ?? 100;
console.log(y);  // 100
    
# ??= 空赋值操作符 ( 不支持 非空赋值 ?.= )
let z = null;
z ??= 300;
console.log(z); // 300

# ?. 空条件表达式
let result = objClass?.name?.length ?? 100;
console.log(result); // 100


```

```csharp
# [ 对象函数调用 ] 通过 ?.() 避免 "xxx is not a function" 错误
let obj = null
obj?.method?.()   // 安全调用：函数/方法


```
