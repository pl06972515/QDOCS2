```csharp
# [ 创建任务 ] 初始状态 Pending 
const task = new Promise((resolve, reject) => {

     resolve("任务完成");           // [ A ][ 任务完成 ] 发送事件 → 触发回调 task.then( ... )  
     reject(new Error("任务失败")); // [ B ][ 任务失败 ] 发送事件 → 触发回调 task.catch( ... ) 

});

------------------------------------------------

# [ 回调函数 ]
{
    
     # [ A ][ 注册回调 ] Promise
     const completedTask = task.then(result => {

     });

     # [ B ][ 注册回调 ] Promise
     const faultedTask = task.catch(error => {

     })
 
     # [ C ][ 注册回调 ] Promise
     const finallyTask = task.finally(() => {

     });

}

# [ 静态方法 ]
Promise.resolve( ... ) -> Promise // Task.FromResult( ... )
Promise.reject(Error) -> Promise  // Task.FromException( ... )
Promise.all([ ... ]) -> Promise   // 等待一组 Promise 完成
    
    
```

#### **[ ℹ️ Task ] 链式调用 **

```csharp
const task = Promise.resolve(100);
task.then(result => {
    
    console.log("成功:", result); // 100
    return result + 50;
    
}).then(result => {

    console.log("成功:", result); // 150
    
}).catch(error => {

    console.error("失败:", error.message);
}).finally(() => {

    console.dir("任务已完成，无论成功或失败。");
});
 

```

#### **[ 🟢 封装 ] XMLHttpRequest**

```csharp
const task = new Promise((resolve, reject) => {

     const xhr = new XMLHttpRequest();
     xhr.open('GET', 'http://192.168.8.113:5000');
     xhr.addEventListener('loadend', function () {

          if (this.status >= 200 && this.status < 300) {
              resolve(this.responseText);
              return;
          }
          reject(new Error("请求失败，状态码：" + this.status));

     });
    xhr.send();

});

task.then(result => {

    console.log("成功:", result);
}).catch(error => {

    console.error("失败:", error.message);
}).finally(() => {

    console.dir("任务已完成，无论成功或失败。");
});

console.log("任务已启动，等待结果...");


```
